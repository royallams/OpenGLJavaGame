#version 330 core

in vec2 pass_textureCoords;
in vec3 surfaceNormal;
in vec3 toLightVector;


out vec4 out_Color;

uniform vec3 lightColor;
uniform sampler2D textureSampler;

void main(){
    vec3 unitNormal = normalize(surfaceNormal);
    vec3 unitLightVector = normalize(toLightVector);

    float dot_prod = dot(unitNormal,unitLightVector);
    float brightness = max(dot_prod, 0);
    vec3 diffuse = brightness * lightColor;
    out_Color = vec4(diffuse,1.0) * texture(textureSampler, pass_textureCoords);// Return the color of the pixel at the coordinate.. sample it at the texture coordinates.
}
